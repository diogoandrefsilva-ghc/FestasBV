CREATE SCHEMA IF NOT EXISTS festasbv;

-- ─── Tabelas de acesso (mesmo padrão do SplitBill) ───────────

CREATE TABLE festasbv.allowed_users (
  email TEXT PRIMARY KEY,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE festasbv.access_requests (
  email TEXT PRIMARY KEY,
  requested_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- ─── Tabela-pai: eventos (1 por ano) ─────────────────────────

CREATE TABLE festasbv.eventos (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  nome TEXT NOT NULL,
  ano INTEGER NOT NULL UNIQUE,
  tesoureiro TEXT NOT NULL,
  arredonda_total BOOLEAN NOT NULL DEFAULT true,
  missao_poupanca NUMERIC(10,2) NOT NULL DEFAULT 0,
  fundo_reserva NUMERIC(10,2) NOT NULL DEFAULT 0
);

-- ─── Filhas (FK ON DELETE CASCADE) ───────────────────────────
-- Nota: 'quem', 'de', 'para', 'membro' ficam como TEXT (nome),
-- porque toda a lógica da app é baseada em nomes. Os membros têm
-- UNIQUE(evento_id, nome) para garantir integridade dentro do ano.

CREATE TABLE festasbv.membros (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  evento_id BIGINT NOT NULL REFERENCES festasbv.eventos(id) ON DELETE CASCADE,
  nome TEXT NOT NULL,
  fator NUMERIC(5,2) NOT NULL DEFAULT 1,
  sobras NUMERIC(10,2) NOT NULL DEFAULT 0,
  outras NUMERIC(10,2) NOT NULL DEFAULT 0,
  UNIQUE (evento_id, nome)
);

CREATE TABLE festasbv.presencas (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  membro_id BIGINT NOT NULL REFERENCES festasbv.membros(id) ON DELETE CASCADE,
  dia TEXT NOT NULL,          -- 'Sex', 'Sáb', …
  ref TEXT NOT NULL,          -- 'Almoço' / 'Jantar'
  UNIQUE (membro_id, dia, ref)
);

CREATE TABLE festasbv.refeicoes_def (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  evento_id BIGINT NOT NULL REFERENCES festasbv.eventos(id) ON DELETE CASCADE,
  data DATE NOT NULL,
  dia TEXT NOT NULL,
  ref TEXT NOT NULL,
  peso NUMERIC(6,3) NOT NULL DEFAULT 0,
  min_meo NUMERIC(10,2) NOT NULL DEFAULT 0,
  min_conv NUMERIC(10,2) NOT NULL DEFAULT 0,
  extra_conv NUMERIC(10,2) NOT NULL DEFAULT 0,
  UNIQUE (evento_id, dia, ref)
);

CREATE TABLE festasbv.despesas (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  evento_id BIGINT NOT NULL REFERENCES festasbv.eventos(id) ON DELETE CASCADE,
  quem TEXT NOT NULL,
  data_desp DATE,
  data_valor DATE,
  descricao TEXT NOT NULL DEFAULT '',
  tipo TEXT NOT NULL,         -- Gerais / Bebidas / Cerveja / Almoço / Jantar / Renda
  valor NUMERIC(10,2) NOT NULL
);

CREATE TABLE festasbv.convidados (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  evento_id BIGINT NOT NULL REFERENCES festasbv.eventos(id) ON DELETE CASCADE,
  membro TEXT NOT NULL,       -- membro MEO responsável
  nome TEXT NOT NULL,
  data DATE,
  dia TEXT NOT NULL,
  ref TEXT NOT NULL,
  pagante BOOLEAN NOT NULL DEFAULT true,
  preco NUMERIC(10,2) NOT NULL DEFAULT 0
);

CREATE TABLE festasbv.mealheiros (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  evento_id BIGINT NOT NULL REFERENCES festasbv.eventos(id) ON DELETE CASCADE,
  quem TEXT NOT NULL,
  data DATE,
  valor NUMERIC(10,2) NOT NULL,
  subtipo TEXT NOT NULL DEFAULT 'lata',   -- lata / sobras_ano_anterior / outros
  descricao TEXT NOT NULL DEFAULT ''
);

CREATE TABLE festasbv.pagamentos (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  evento_id BIGINT NOT NULL REFERENCES festasbv.eventos(id) ON DELETE CASCADE,
  de TEXT NOT NULL,
  para TEXT NOT NULL,
  valor NUMERIC(10,2) NOT NULL,
  ref TEXT NOT NULL DEFAULT '',
  data DATE
);

-- Índices nas FKs (acelera embedded resources e deletes em cascata)
CREATE INDEX ON festasbv.membros (evento_id);
CREATE INDEX ON festasbv.presencas (membro_id);
CREATE INDEX ON festasbv.refeicoes_def (evento_id);
CREATE INDEX ON festasbv.despesas (evento_id);
CREATE INDEX ON festasbv.convidados (evento_id);
CREATE INDEX ON festasbv.mealheiros (evento_id);
CREATE INDEX ON festasbv.pagamentos (evento_id);
